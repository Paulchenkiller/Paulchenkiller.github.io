Kontakt

Ansonsten versuche es mit diesem Perlcode:

#!/usr/bin/perl

use strict;
use warnings;

my @contact = qw(86 103 114 117 119 101 104 001 103 117 001 100 113 101 104 001 112 103 114 001 71 47 79 99 107 108 35 001 61 47 43);
foreach my $c ( @contact ) {
  if ( $c == 001 ) {
    print " ";
  }
  else {
    # da ist wohl was falsch.....
    print chr( $c );
  }
}

sub key {
  my @key =
     qw(99 104 114 40 110 36 99 37 50 33 61 48 63 36 99 45 61 50 58 36 99 110 41);
  foreach my $k ( @key ) {
    if ( $k == 110 ) {
      print " ";
    }
    else {
      print chr( $k );
    }
  }
}
